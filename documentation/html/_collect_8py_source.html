213 def find_dimension(p):
214  """
215  Divide the total process p into p_rows Ã— p_cols
216 
217  Parameters
218  ----------
219  p: integer
220  The size of the all processes
221 
222  Returns
223  ----------
224  p_rows: integer
225  p_cols: intger
226 
227  Example
228  ----------
229  >>> m,n = find_dimension(6)
230  >>>print(m)
231  >>> 2
232  >>>print(n)
233  >>> 3
234  """
235  min_gap = p
236  top = math.floor(math.sqrt(p) + 1)
237  for i in range(1, top + 1):
238  if(p % i == 0):
239  gap = math.floor(abs(p/i-i))
240  if(gap < min_gap):
241  min_gap = gap
242  p_rows = i
243  p_cols = math.floor(p/i)
244 
245  return p_rows, p_cols

248 def setup_partition_grid(p, imax, jmax):
249  """
250  This function is used to do grid decomposition
251 
252  Parameters
253  ----------
254  p: integer
255  The size of the all processes
256  imax: integer
257  The rows of the matrix
258  jmax: integer
259  The columns of the matrix
260 
261 
262  Returns
263  ----------
264  row_start: list
265  records the row start number in whole matrix for every process
266  row_process_chunk: list
267  records the number of rows that every process needs to deal with
268  col_start: list
269  records the column start number in whole matrix for every process
270  col_process_chunk: list
271  records the the number of colunmns that every process needs to
272  deal with
273  """
274  p_rows, p_cols = find_dimension(p)
275  row_start = list(range(p_rows))
276  col_start = list(range(p_cols))
277  row_process_chunk = list(range(p_rows))
278  col_process_chunk = list(range(p_cols))
279  # Becasue the first row and last row are boundray, the number of
280  # rows to deal with for all processes are the total rows minus 2
281  rows_left = imax - 2
282  row_start[0] = 0
283  for n in range(0, p_rows - 1):
284  row_assigned = math.floor(rows_left / (p_rows - n))
285  rows_left -= row_assigned
286  row_start[n + 1] = row_start[n] + row_assigned
287  row_process_chunk[n] = row_assigned
288  row_process_chunk[p_rows - 1] = imax - 2 - row_start[p_rows - 1]
289 
290  # add the top boundray, process 0's rows start number is still 0
291  # the others must add 1
292  for i in range(1, p_rows):
293  row_start[i] = row_start[i] + 1
294 
295  # because the top boundray and the boundary
296  # add 1 to the process_chunk
297  row_process_chunk[0] = row_process_chunk[0] + 1
298  row_process_chunk[p_rows - 1] = row_process_chunk[p_rows - 1] + 1
299 
300  # This is same to rows
301  col_left = jmax - 2
302  col_start[0] = 0
303  for n in range(0, p_cols - 1):
304  col_assigned = math.floor(col_left / (p_cols - n))
305  col_left -= col_assigned
306  col_start[n + 1] = col_start[n] + col_assigned
307  col_process_chunk[n] = col_assigned
308  col_process_chunk[p_cols - 1] = jmax - 2 - col_start[p_cols - 1]
309 
310  # add the left boundray and right boundray
311  # the others must add 1
312  for i in range(1, p_cols):
313  col_start[i] = col_start[i] + 1
314 
315  # because the left boundary and the last row has the boundary
316  # add 1 to the process_chunk
317  col_process_chunk[0] = col_process_chunk[0] + 1
318  col_process_chunk[p_cols - 1] = col_process_chunk[p_cols - 1] + 1
319 
320  return row_start, row_process_chunk, col_start, col_process_chunk
